Nginx 启动命令

docker run -d --name nginx -p 80:80 -v /ss-data/nginx/nginx.conf:/etc/nginx/nginx.conf -v /ss-data/nginx/logs:/var/log/nginx -v /ss-data/nginx/html:/usr/share/nginx/html -v /ss-data/nginx/conf:/etc/nginx/conf.d --privileged=true nginx

1. 开启ssh服务

systemctl enable ssh.service

2. 离线安装

tar -zxvf docker-24.0.2.tgz
sudo mv docker/* /usr/bin/

3. 添加开机启动

# 添加开机启动配置
sudo vim /usr/lib/systemd/system/docker.service

文件内容

[Unit]
Description=Docker Application Container Engine
Documentation=https://docs.docker.com
After=network-online.target firewalld.service
Wants=network-online.target
[Service]
Type=notify
ExecStart=/usr/bin/dockerd
ExecReload=/bin/kill -s HUP $MAINPID
LimitNOFILE=infinity
LimitNPROC=infinity
TimeoutStartSec=0
Delegate=yes
KillMode=process
Restart=on-failure
StartLimitBurst=3
StartLimitInterval=60s
[Install]
WantedBy=multi-user.target

添加执行权限

chmod +x /usr/lib/systemd/system/docker.service
sudo systemctl daemon-reload

启动Docker

sudo systemctl start docker

配置开机启动

sudo systemctl enable docker

如果上面的命令执行不成功，可通过以下方式实现

echo "systemctl start docker" > /etc/rc.local
chmod +x /etc/rc.local

4. Portainer启动命令

# 启动 docker 镜像
docker run -p 9000:9000 -p 8000:8000 --name portainer \
--restart=always \
-v /var/run/docker.sock:/var/run/docker.sock \
-v /mnt/docker_data/portainer/data:/data \
-d portainer/portainer-ce:2.18.3-alpine

Nginx配置

# 将容器nginx.conf文件复制到宿主机
docker cp d576bf1439d8:/etc/nginx/nginx.conf   /mnt/docker_data/nginx/conf/nginx.conf

# 将容器conf.d文件夹下内容复制到宿主机
docker cp d576bf1439d8:/etc/nginx/conf.d   /mnt/docker_data/nginx/conf/conf.d

# 将容器中的html文件夹复制到宿主机
docker cp d576bf1439d8:/usr/share/nginx/html  /mnt/docker_data/nginx/static

Portainer Stacks

basic

version: "3.5"
networks:
  default:
    external: true
    name: data
    
services: 
  # Redis 服务
  redis:
    image: redis:7.0.11-alpine
    container_name: redis
    restart: always
    ports:
      - '6379:6379'
    volumes:
        - /mnt/docker_data/redis/data:/data
        - /mnt/docker_data/redis/config/redis.conf:/usr/local/etc/redis/redis.conf
        - /mnt/docker_data/redis/log:/logs
    command:
      redis-server /usr/local/etc/redis/redis.conf
  
  #Nginx 服务
  nginx:
    image: nginx:1.25.0-alpine
    container_name: nginx
    restart: always
    ports: 
      - '80:80'
    volumes:
      - /mnt/docker_data/nginx/conf/nginx.conf:/etc/nginx/nginx.conf
      - /mnt/docker_data/nginx/conf/conf.d:/etc/nginx/conf.d
      - /mnt/docker_data/nginx/static:/usr/share/nginx/html

  # RocketMQ NameServer 服务
  namesrv:
    image: apache/rocketmq:5.1.1
    container_name: rmqnamesrv
    restart: always
    ports:
      - 9876:9876
    volumes:
      - /mnt/docker_data/rocketmq/namesrv/logs:/home/rocketmq/logs
    command: sh mqnamesrv
    
  # RocketMQ Broker 服务
  broker:
    image: apache/rocketmq:5.1.1
    container_name: rmqbroker
    restart: always
    ports:
      - 10909:10909
      - 10911:10911
      - 10912:10912
    volumes:
      - /mnt/docker_data/rocketmq/broker/logs:/home/rocketmq/logs
      - /mnt/docker_data/rocketmq/broker/store:/home/rocketmq/store
      - /mnt/docker_data/rocketmq/broker/conf/broker.conf:/home/rocketmq/rocketmq-5.1.1/conf/broker.conf
    command: sh mqbroker -n namesrv:9876 -c ../conf/broker.conf
    depends_on:
      - namesrv

  # RocketMQ 仪表盘
  rocketmq-dashboard:
    image: apacherocketmq/rocketmq-dashboard:latest
    container_name: rocketmq-dashboard
    restart: always
    depends_on:
      - namesrv
    ports:
      - '28080:8080'
    environment:
      - JAVA_OPTS=-Drocketmq.namesrv.addr=192.168.32.200:9876

  # Postgresql 服务
  postgres:
    image: postgres:15.2-alpine
    container_name: postgres
    restart: always
    ports:
      - 5432:5432
    environment:
      - POSTGRES_PASSWORD=geovisPostgres
      - ALLOW_IP_RANGE=0.0.0.0/0
      - TZ=Asia/Shanghai
      - PGTZ=Asia/Shanghai
    volumes:
      - /mnt/docker_data/postgres/data:/var/lib/postgresql/data
    logging: 
      driver: "json-file"
      options: 
        max-size: "10m"       
    # MySQL 服务  
  mysql:  
    image: mysql:5.7 
    container_name: mysql   
    restart: always  
    ports:  
      - 3306:3306  
    environment:  
      - MYSQL_ROOT_PASSWORD=root  

      - MYSQL_PASSWORD=root

    volumes:  
      - /mnt/docker_data/mysql/data:/var/lib/mysql
